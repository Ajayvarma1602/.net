using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace CustomerLibrary
{
    public class Customer
    {
        public int CustomerID { get; set; }
        public string FirstName { get; set; }
        public string LastName { get; set; }
        public string Location { get; set; }
        
    }//customer class closes

    public class CustomerCollection
    {
        Customer[] customerList = null;
        int size;
        static int i; //default value for i is zero it will used as index of the array

        public CustomerCollection(int s=1)
        {
            customerList = new Customer[s];
            size = s;
        }

        public int Size
        {
            get { return size; }
        }

        public void Insert()
        {
            if (i >= size)
            {
                Console.WriteLine("No space to add new customer..");
            }
            else
            {
                Customer c = new Customer();
                Console.Write("Enter Customer Id = ");
                c.CustomerID = Convert.ToInt32(Console.ReadLine());
                Console.Write("Enter First Name = ");
                c.FirstName = Console.ReadLine();
                Console.Write("Enter Last Name = ");
                c.LastName = Console.ReadLine();
                Console.Write("Enter Location = ");
                c.Location = Console.ReadLine();

                //add this customer object in the array at the given index
                customerList[i] = c;
                i++;
            }
        }

        void DisplayCustomer(Customer c)
        {
            Console.WriteLine("Customer Id : " + c.CustomerID);
            Console.WriteLine("First Name  : " + c.FirstName);
            Console.WriteLine("Last Name   : " + c.LastName);
            Console.WriteLine("Location    : " + c.Location);
        }

        public void ShowCustomerById(int id)
        {
            Customer c = null;
            if (i == 0)
            {
                Console.WriteLine("No customer to display.. add a new customer");
            }
            else
            {
                foreach (Customer customer in customerList)
                {
                    if (customer != null)
                    {
                        if (customer.CustomerID == id)
                        {
                            c = customer;
                            break;
                        }
                    }
                }

                if (c == null)
                {
                    Console.WriteLine("Customer with given ID not found");
                }
                else
                {
                    DisplayCustomer(c);
                }
            }
        }

        public void ShowAllCustomers()
        {
            if (i == 0)
            {
                Console.WriteLine("No customers to display.. add a new customer");
            }
            else
            {
                Console.WriteLine("Details of all customers:");
                Console.WriteLine("--------------------------");
                foreach (Customer c in customerList)
                {
                    if (c != null)
                    {
                        DisplayCustomer(c);
                        Console.WriteLine("--------------------------");
                    }
                }
            }
        }


        public void Update(int id)
        {
            Customer c = null;
            foreach (Customer customer in customerList)
            {
                if (customer.CustomerID == id)
                {
                    c = customer;
                    break;
                }
            }

            if (c == null)
            {
                Console.WriteLine("Customer with given ID not found");
            }
            else
            {
                Console.WriteLine("Existing Details of the customer :");
                DisplayCustomer(c);

                Console.WriteLine("Provide the new details for this customer : ");
                Console.Write("Enter First Name = ");
                c.FirstName = Console.ReadLine();
                Console.Write("Enter Last Name = ");
                c.LastName = Console.ReadLine();
                Console.Write("Enter Location = ");
                c.Location = Console.ReadLine();

                Console.WriteLine("Changes Updated");
            }
        }

        public void Delete(int id)
        {
            Customer c = null;
            foreach (Customer customer in customerList)
            {
                if (customer.CustomerID == id)
                {
                    c = customer;
                    break;
                }
            }

            if (c == null)
            {
                Console.WriteLine("Customer with given ID not found");
            }
            else
            {
                Console.WriteLine("Existing Details of the customer :");
                DisplayCustomer(c);

                Console.Write("Do you wish to delete this record (y or n) : ");
                char answer = Convert.ToChar(Console.ReadLine());

                if(answer=='y' || answer=='Y')
                {
                    c = null;
                    Console.WriteLine("Record deleted.");
                }

            }
        }

    }

}

Program.cs

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

using CustomerLibrary;

namespace CustomerConsoleApp
{
    class Program
    {
        static void Main(string[] args)
        {
            CustomerCollection customers = new CustomerCollection(5);

            int choice;

            int id;

            do
            {
                Console.WriteLine("*** Customer Operations ***");
                Console.WriteLine("1. Show All Customers");
                Console.WriteLine("2. Show Customer By Id");
                Console.WriteLine("3. Add New Customer");
                Console.WriteLine("4. Update Customer");
                Console.WriteLine("5. Delete Customer");
                Console.WriteLine("0. Exit");

                Console.Write("Enter the operation to perform : ");
                choice = Convert.ToInt32(Console.ReadLine());

                switch(choice)
                {
                    case 1:
                        customers.ShowAllCustomers();
                        break;
                    case 2:
                        Console.Write("Enter the Customer Id = ");
                        id = Convert.ToInt32(Console.ReadLine());
                        customers.ShowCustomerById(id);
                        break;
                    case 3:
                        customers.Insert();
                        break;
                    case 4:
                        Console.Write("Enter the Customer Id = ");
                        id = Convert.ToInt32(Console.ReadLine());
                        customers.Update(id);
                        break;
                    case 5:
                        Console.Write("Enter the Customer Id = ");
                        id = Convert.ToInt32(Console.ReadLine());
                        customers.Delete(id);
                        break;
                    default:
                        if(choice!=0)
                        {
                            Console.WriteLine("Invalid choice..try again");
                        }
                        break;
                 }
                Console.WriteLine("-----------------------------------");
               } while (choice != 0);
        }
    }
}
